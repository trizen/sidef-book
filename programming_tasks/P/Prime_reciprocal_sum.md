# [Prime reciprocal sum][1]

```ruby
var A075442 = Enumerator({|callback|
    var sum = 0
    loop {
        var p = next_prime(ceil(1/(1-sum)))
        sum += 1/p
        callback(p)
    }
})

A075442.first(15).each_kv {|k,n|
    var s = Str(n)
    s = "#{s.first(20)}..#{s.last(20)} (#{s.len} digits)" if (s.len > 50)
    say "#{'%2d' % k+1}: #{s}"
}
```
