# [Padovan n-step number sequences][1]

```ruby
func padovan(N) {
    Enumerator({|callback|
        var n = 2
        var pn = [1, 1, 1]
        loop {
            pn << sum(pn[n-N .. (n++-1) -> grep { _ >= 0 }])
            callback(pn[-4])
        }
    })
}

for n in (2..8) {

}
```
